// Get dynamic data from SetupReference Sheet
function getSetupReferences() {
    try {
      const ss = SpreadsheetApp.getActiveSpreadsheet();
      const sheet = ss.getSheetByName('SetupReference Sheet');
      
      if (!sheet) return {
        parentPDArrays: [],
        liquidityTypes: [],
        fvgTypes: [],
        secondaryPatterns: []
      };
      
      // Get Parent PD Arrays (Column A)
      const parentPDArrays = sheet.getRange("A3:A" + sheet.getLastRow())
        .getValues()
        .filter(row => row[0] !== "")
        .map(row => row[0]);
      
      // Get Liquidity Types (Column B)
      const liquidityTypes = sheet.getRange("B3:B" + sheet.getLastRow())
        .getValues()
        .filter(row => row[0] !== "")
        .map(row => row[0]);
      
      // Get FVG Types (Column C and some from Column G)
      const keyFvgTypes = sheet.getRange("C3:C" + sheet.getLastRow())
        .getValues()
        .filter(row => row[0] !== "")
        .map(row => row[0]);
      
      const secondaryFvgTypes = sheet.getRange("G3:G" + sheet.getLastRow())
        .getValues()
        .filter(row => row[0] !== "" && row[0].includes("FVG"))
        .map(row => row[0]);
      
      const fvgTypes = [...new Set([...keyFvgTypes, ...secondaryFvgTypes])];
      
      // Get Secondary Patterns (Column E)
      const secondaryPatterns = sheet.getRange("E3:E" + sheet.getLastRow())
        .getValues()
        .filter(row => row[0] !== "")
        .map(row => row[0]);
      
      return {
        parentPDArrays,
        liquidityTypes,
        fvgTypes,
        secondaryPatterns
      };
    } catch (error) {
      console.error("Error in getSetupReferences:", error);
      return {
        parentPDArrays: [],
        liquidityTypes: [],
        fvgTypes: [],
        secondaryPatterns: []
      };
    }

  function doGet() {
    // Get dynamic data from the SetupReference Sheet
    const setupData = getSetupReferences();
    
    // Create template and pass data to it
    const template = HtmlService.createTemplateFromFile('SetupAnalysisForm');
    template.parentPDArrays = setupData.parentPDArrays;
    template.liquidityTypes = setupData.liquidityTypes;
    template.fvgTypes = setupData.fvgTypes;
    template.secondaryPatterns = setupData.secondaryPatterns;
    
    return template
      .evaluate()
      .setTitle('Trading Setup Statistical Analysis')
      .setFaviconUrl('https://ssl.gstatic.com/docs/spreadsheets/favicon3.ico');
  }

  function recordSetupStatistics(formData) {
    try {
      const ss = SpreadsheetApp.getActiveSpreadsheet();
      const sheet = ss.getSheetByName('Data Entry Form');
      
      if (!sheet) {
        return {
          success: false,
          message: "Data Entry Form sheet not found"
        };
      }
      
      // Log the submitted data for debugging
      logMessage(`Statistical analysis submitted for date: ${formData.tradeDate}, time: ${formData.tradeTime}`);
      
      // Get all data from the sheet
      const dataRange = sheet.getDataRange();
      const data = dataRange.getValues();
      
      // Parse form date using existing utility function
      const formDate = parseAndNormalizeDate(formData.tradeDate, null, Session.getScriptTimeZone());
      const formDateStr = Utilities.formatDate(formDate, Session.getScriptTimeZone(), 'yyyy-MM-dd');
      
      // Parse form time
      let formHours = 0;
      let formMinutes = 0;
      if (formData.tradeTime) {
        const timeParts = formData.tradeTime.split(':');
        formHours = parseInt(timeParts[0], 10);
        formMinutes = parseInt(timeParts[1], 10);
      }
      
      logMessage(`Looking for trade on ${formDateStr} at ${formHours}:${formMinutes}`);
      
      // Find matching row
      let targetRow = -1;
      const dateCol = 0; // Column A (index 0)
      const timeCol = 4; // Column E (index 4)
      
      // Start from row 3 (skipping headers)
      for (let i = 2; i < data.length; i++) {
        const rowDate = data[i][dateCol];
        const rowTime = data[i][timeCol];
        
        // Skip empty rows
        if (!rowDate) continue;
        
        // Normalize row date for comparison
        const normalizedRowDate = normalizeDate(rowDate);
        if (!normalizedRowDate) {
          logMessage(`Row ${i+1}: Invalid date format: ${rowDate}`);
          continue;
        }
        
        // Format row date for string comparison
        const rowDateStr = Utilities.formatDate(normalizedRowDate, Session.getScriptTimeZone(), 'yyyy-MM-dd');
        
        // Extract hours and minutes from row time
        let rowHours = 0;
        let rowMinutes = 0;
        
        if (rowTime instanceof Date) {
          rowHours = rowTime.getHours();
          rowMinutes = rowTime.getMinutes();
        } else if (typeof rowTime === 'string') {
          const timeParts = rowTime.split(':');
          if (timeParts.length >= 2) {
            rowHours = parseInt(timeParts[0], 10);
            rowMinutes = parseInt(timeParts[1], 10);
          }
        } else if (typeof rowTime === 'number') {
          rowHours = Math.floor(rowTime);
          rowMinutes = Math.round((rowTime - rowHours) * 60);
        }
        
        // Compare dates and times
        if (rowDateStr === formDateStr && 
            rowHours === formHours && 
            Math.abs(rowMinutes - formMinutes) <= 1) { // Allow 1 minute tolerance
          targetRow = i + 1;
          logMessage(`Match found at row ${targetRow}`);
          break;
        }
      }
      
      if (targetRow === -1) {
        // If no match found, look through all rows of the correct date
        let possibleMatches = [];
        for (let i = 2; i < data.length; i++) {
          const rowDate = data[i][dateCol];
          if (!rowDate) continue;
          
          const normalizedRowDate = normalizeDate(rowDate);
          if (!normalizedRowDate) continue;
          
          const rowDateStr = Utilities.formatDate(normalizedRowDate, Session.getScriptTimeZone(), 'yyyy-MM-dd');
          
          if (rowDateStr === formDateStr) {
            // Found a row with matching date
            const rowTime = data[i][timeCol];
            let timeDisplay = rowTime;
            if (rowTime instanceof Date) {
              timeDisplay = Utilities.formatDate(rowTime, Session.getScriptTimeZone(), 'HH:mm:ss');
            }
            possibleMatches.push(`Row ${i+1}: Time=${timeDisplay}`);
          }
        }
        
        // Build detailed error message
        let errorMessage = `No matching trade found for date ${formDateStr} and time ${formData.tradeTime}. `;
        if (possibleMatches.length > 0) {
          errorMessage += `Found trades on ${formDateStr} with these times: ${possibleMatches.join(", ")}`;
        } else {
          errorMessage += `No trades found on ${formDateStr}. Please check that the date and time match exactly with your Data Entry Form.`;
        }
        
        return {
          success: false,
          message: errorMessage
        };
      }
      
      // Now that we found the correct row, store the data in columns AA-AJ
      logMessage(`Processing trade at row ${targetRow} with form data`);
      
      try {
        // First, add headers if they don't exist (only once)
        if (sheet.getRange("AA1").getValue() === "") {
          // Setup Header 1 for Pattern Structure & Correlations (columns AA-AD)
          sheet.getRange("AA1").setValue("PATTERN STRUCTURE & CORRELATIONS").setFontWeight("bold");
          sheet.getRange("AA1:AD1").merge().setBackground("#4285f4").setFontColor("white");
          
          // Sub-headers for first section
          sheet.getRange("AA2").setValue("Pattern Components");
          sheet.getRange("AB2").setValue("FVG Types");
          sheet.getRange("AC2").setValue("Market Context");
          sheet.getRange("AD2").setValue("Pattern Evolution");
          sheet.getRange("AA2:AD2").setBackground("#e6e6e6").setFontWeight("bold");
          
          // Setup Header 2 for Sequence Analysis (columns AE-AH)
          sheet.getRange("AE1").setValue("PRICE ACTION SEQUENCE ANALYSIS").setFontWeight("bold");
          sheet.getRange("AE1:AH1").merge().setBackground("#4285f4").setFontColor("white");
          
          // Sub-headers for second section
          sheet.getRange("AE2").setValue("Sequence Steps");
          sheet.getRange("AF2").setValue("Liquidity Analysis");
          sheet.getRange("AG2").setValue("News Impact");
          sheet.getRange("AH2").setValue("Key Observations");
          sheet.getRange("AE2:AH2").setBackground("#e6e6e6").setFontWeight("bold");
          
          // Setup Header 3 for Additional Data (columns AI-AJ)
          sheet.getRange("AI1").setValue("ADDITIONAL INFORMATION").setFontWeight("bold");
          sheet.getRange("AI1:AJ1").merge().setBackground("#4285f4").setFontColor("white");
          
          // Sub-headers for third section
          sheet.getRange("AI2").setValue("Correlation Notes");
          sheet.getRange("AJ2").setValue("Analysis Timestamp");
          sheet.getRange("AI2:AJ2").setBackground("#e6e6e6").setFontWeight("bold");
        }
        
        // Prepare data for Section 1: Pattern Structure & Correlations
        
        // Pattern Components - Column AA
        let patternComponents = [];
        if (formData.primarySetup) patternComponents.push(formData.primarySetup);
        if (formData.parentPDArrayPresent === "Yes" && formData.parentPDArray) {
          patternComponents.push(`PD: ${formData.parentPDArray}`);
        }
        if (Array.isArray(formData.secondaryPatterns) && formData.secondaryPatterns.length > 0) {
          patternComponents.push(`Secondary: ${formData.secondaryPatterns.join(", ")}`);
        }
        
        // FVG Types - Column AB
        let fvgTypesStr = "";
        if (Array.isArray(formData.fvgTypes) && formData.fvgTypes.length > 0) {
          fvgTypesStr = formData.fvgTypes.join(", ");
        }
        
        // Market Context - Column AC
        let marketContextArr = [];
        if (formData.newsPresent) marketContextArr.push(formData.newsPresent);
        if (formData.newsPresent !== "No News Impact" && formData.newsType) {
          marketContextArr.push(`${formData.newsType} (Impact: ${formData.newsImpact || "N/A"})`);
        }
        if (formData.liquidityTaken) {
          marketContextArr.push(`Liquidity: ${formData.liquidityTaken} (${formData.liquidityTakenResult || "N/A"})`);
        }
        let marketContextStr = marketContextArr.join("\n");
        
        // Pattern Evolution - Column AD
        let evolutionArr = [];
        if (Array.isArray(formData.midTradeDevelopments) && formData.midTradeDevelopments.length > 0) {
          evolutionArr.push(formData.midTradeDevelopments.join("\n"));
        }
        if (formData.adaptationType) {
          evolutionArr.push(`Adaptation: ${formData.adaptationType}`);
        }
        let evolutionStr = evolutionArr.join("\n");
        
        // Prepare data for Section 2: Sequence Analysis
        
        // Sequence Steps - Column AE
        let sequenceStr = "";
        if (formData.sequenceData && Array.isArray(formData.sequenceData)) {
          // Format sequence data nicely with arrows and details
          sequenceStr = formData.sequenceData
            .filter(item => item.type) // Only include items with a type
            .map((item, index) => {
              const timeStr = item.time ? ` (${item.time})` : "";
              const sessionStr = item.session ? ` [${item.session}]` : "";
              return `${index+1}. ${item.type}${timeStr}${sessionStr}`;
            })
            .join("\n ");
        }
        
        // Liquidity Analysis - Column AF
        let liquidityAnalysisArr = [];
        if (formData.liquidityQuality) {
          liquidityAnalysisArr.push(`Quality: ${formData.liquidityQuality}/3`);
        }
        if (formData.liquidityTakenResult) {
          liquidityAnalysisArr.push(`Result: ${formData.liquidityTakenResult}`);
        }
        let liquidityAnalysisStr = liquidityAnalysisArr.join("\n");
        
        // News Impact - Column AG
        let newsImpactStr = "";
        if (formData.newsPresent && formData.newsPresent !== "No News Impact") {
          let newsComponents = [];
          newsComponents.push(formData.newsPresent);
          if (formData.newsType) newsComponents.push(`Type: ${formData.newsType}`);
          if (formData.newsImpact) newsComponents.push(`Impact Level: ${formData.newsImpact}/5`);
          if (formData.newsProximity) newsComponents.push(`Proximity: ${formData.newsProximity} mins`);
          newsImpactStr = newsComponents.join("\n");
        } else {
          newsImpactStr = "No significant news impact";
        }
        
        // Key Observations - Column AH
        let keyObservationsStr = formData.keyObservations || "";
        
        // Prepare data for Section 3: Additional Information
        
        // Correlation Notes - Column AI
        let correlationNotesStr = formData.correlationNotes || "";
        
        // Analysis Timestamp - Column AJ
        let timestamp = new Date().toISOString();
        
        // Write all data to the correct cells
        
        // Section 1: Pattern Structure & Correlations (AA-AD)
        sheet.getRange(targetRow, 27).setValue(patternComponents.join("\n")); // AA (27)
        sheet.getRange(targetRow, 28).setValue(fvgTypesStr);                  // AB (28)
        sheet.getRange(targetRow, 29).setValue(marketContextStr);             // AC (29)
        sheet.getRange(targetRow, 30).setValue(evolutionStr);                 // AD (30)
        
        // Section 2: Sequence Analysis (AE-AH)
        sheet.getRange(targetRow, 31).setValue(sequenceStr);                  // AE (31)
        sheet.getRange(targetRow, 32).setValue(liquidityAnalysisStr);         // AF (32)
        sheet.getRange(targetRow, 33).setValue(newsImpactStr);                // AG (33)
        sheet.getRange(targetRow, 34).setValue(keyObservationsStr);           // AH (34)
        
        // Section 3: Additional Information (AI-AJ)
        sheet.getRange(targetRow, 35).setValue(correlationNotesStr);          // AI (35)
        sheet.getRange(targetRow, 36).setValue(timestamp);                    // AJ (36)
        
        // Format cells for readability
        sheet.getRange(targetRow, 27, 1, 10).setVerticalAlignment("top").setWrap(true);
        
        return {
          success: true,
          message: `Statistical analysis saved for trade on ${formDateStr} at ${formData.tradeTime}. Data stored in columns AA-AJ of your Data Entry Form.`
        };
        
      } catch (storageError) {
        logMessage(`Error storing statistical data: ${storageError.toString()}`);
        
        return {
          success: false,
          message: `Found the trade but failed to store data: ${storageError.toString()}`
        };
      }
      
    } catch (error) {
      logMessage(`Error in recordSetupStatistics: ${error.toString()}`);
      return {
        success: false,
        message: `Error: ${error.toString()}`
      };
    }
  }
