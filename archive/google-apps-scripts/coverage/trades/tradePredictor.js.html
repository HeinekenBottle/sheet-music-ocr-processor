/**
 * Predicts the likelihood of a winning trade based on input parameters
 * using a logistic regression approach.
 * 
 * @param {Object} tradeFeatures - Object containing features for prediction
 * @returns {Object} Prediction result with probability and classification
 */
function predictTradeOutcome(tradeFeatures) {
    try {
      // Get model coefficients - either from stored values or train new model
      const coefficients = getModelCoefficients() || trainModel();
      
      if (!coefficients || !coefficients.weights) {
        throw new Error("Unable to load or train prediction model");
      }
      
      // Prepare features in the correct format
      const features = normalizeFeatures(tradeFeatures);
      
      // Calculate probability using logistic regression
      const probability = calculateProbability(features, coefficients);
      
      // Determine probability tier
      let probabilityTier;
      if (probability >= 0.7) {
        probabilityTier = 'Excellent';
      } else if (probability >= 0.6) {
        probabilityTier = 'Good';
      } else if (probability >= 0.5) {
        probabilityTier = 'Fair';
      } else {
        probabilityTier = 'Poor';
      }
      
      // Return prediction results
      return {
        calculatedProbability: (probability * 100).toFixed(1) + '%',
        probabilityTier: probabilityTier,
        features: features,
        modelVersion: coefficients.version || '1.0',
        predictedOutcome: probability >= 0.5 ? 'win' : 'loss'
      };
    } catch (e) {
      Logger.log('Error in predictTradeOutcome: ' + e.toString());
      
      // Return fallback values
      return {
        calculatedProbability: '50.0%',
        probabilityTier: 'Fair',
        features: {},
        modelVersion: 'fallback',
        predictedOutcome: 'unknown',
        error: e.toString()
      };
    }
  }
  
  /**
   * Normalizes input features to prepare for prediction
   * 
   * @param {Object} rawFeatures - Raw feature values
   * @returns {Object} Normalized features
   */
  function normalizeFeatures(rawFeatures) {
    // Define expected features and default values
    const defaultFeatures = {
      qualityScore: 3.5,
      previousResult: 'none',  // 'win', 'loss', or 'none'
      liquidityResistance: 'low', // 'low', 'moderate', 'high'
      priceClarity: 'clear',  // 'clear', 'mixed', 'unclear'
      tradeCount: 1,
      marketVolatility: 'normal', // 'low', 'normal', 'high'
      riskReward: 'good',  // 'poor', 'fair', 'good', 'excellent'
      timeOfDay: 0,  // Hour (0-23)
      dayOfWeek: 0,  // 0 = Sunday, 1 = Monday, etc.
    };
    
    // Start with defaults
    const features = {...defaultFeatures};
    
    // Override with provided values
    Object.keys(rawFeatures).forEach(key => {
      if (features.hasOwnProperty(key)) {
        features[key] = rawFeatures[key];
      }
    });
    
    // Convert categorical features to numerical
    const normalizedFeatures = {};
    
    // Quality score (already numerical)
    normalizedFeatures.qualityScore = parseFloat(features.qualityScore) || 3.5;
    
    // Previous result: 0 = none, 1 = win, -1 = loss
    if (typeof features.previousResult === 'string') {
      const result = features.previousResult.toLowerCase();
      normalizedFeatures.previousResult = result === 'win' ? 1 : (result === 'loss' ? -1 : 0);
    } else {
      normalizedFeatures.previousResult = 0;
    }
    
    // Liquidity resistance: 0 = low, 0.5 = moderate, 1 = high
    if (typeof features.liquidityResistance === 'string') {
      const resistance = features.liquidityResistance.toLowerCase();
      normalizedFeatures.liquidityResistance = resistance === 'high' ? 1 : (resistance === 'moderate' ? 0.5 : 0);
    } else {
      normalizedFeatures.liquidityResistance = 0;
    }
    
    // Price clarity: 0 = clear, 0.5 = mixed, 1 = unclear
    if (typeof features.priceClarity === 'string') {
      const clarity = features.priceClarity.toLowerCase();
      normalizedFeatures.priceClarity = clarity === 'unclear' ? 1 : (clarity === 'mixed' ? 0.5 : 0);
    } else {
      normalizedFeatures.priceClarity = 0;
    }
    
    // Trade count (already numerical)
    normalizedFeatures.tradeCount = parseInt(features.tradeCount) || 1;
    
    // Market volatility: 0 = low, 0.5 = normal, 1 = high
    if (typeof features.marketVolatility === 'string') {
      const volatility = features.marketVolatility.toLowerCase();
      normalizedFeatures.marketVolatility = volatility === 'high' ? 1 : (volatility === 'normal' ? 0.5 : 0);
    } else {
      normalizedFeatures.marketVolatility = 0.5;
    }
    
    // Risk/reward: 0 = poor, 0.33 = fair, 0.66 = good, 1 = excellent
    if (typeof features.riskReward === 'string') {
      const riskReward = features.riskReward.toLowerCase();
      if (riskReward === 'excellent') normalizedFeatures.riskReward = 1;
      else if (riskReward === 'good') normalizedFeatures.riskReward = 0.66;
      else if (riskReward === 'fair') normalizedFeatures.riskReward = 0.33;
      else normalizedFeatures.riskReward = 0;
    } else {
      normalizedFeatures.riskReward = 0.66; // Default to 'good'
    }
    
    // Time of day effects (0-23 hours)
    const timeOfDay = parseInt(features.timeOfDay) || 0;
    normalizedFeatures.timeOfDay = timeOfDay / 24; // Normalize to 0-1
    
    // Day of week (0-6)
    const dayOfWeek = parseInt(features.dayOfWeek) || 0;
    normalizedFeatures.dayOfWeek = dayOfWeek / 6; // Normalize to 0-1
    
    // Add an intercept term
    normalizedFeatures.intercept = 1.0;
    
    return normalizedFeatures;
  }
  
  /**
   * Calculate probability using logistic regression formula
   * P(y=1) = 1 / (1 + e^(-z)) where z = w0 + w1*x1 + w2*x2 + ... + wn*xn
   * 
   * @param {Object} features - Normalized feature values
   * @param {Object} coefficients - Model coefficients
   * @returns {number} Probability between 0 and 1
   */
  function calculateProbability(features, coefficients) {
    // Linear combination
    let z = 0;
    
    // Add contribution from each feature
    Object.keys(coefficients.weights).forEach(feature => {
      if (features.hasOwnProperty(feature)) {
        z += coefficients.weights[feature] * features[feature];
      }
    });
    
    // Apply logistic function
    return 1 / (1 + Math.exp(-z));
  }
  
  /**
   * Get stored model coefficients from properties or cache
   * 
   * @returns {Object|null} Model coefficients or null if not found
   */
  function getModelCoefficients() {
    try {
      // Try to get from properties
      const props = PropertiesService.getScriptProperties();
      const modelJson = props.getProperty('PREDICTION_MODEL_COEFFICIENTS');
      
      if (modelJson) {
        return JSON.parse(modelJson);
      }
      
      return null;
    } catch (e) {
      Logger.log('Error getting model coefficients: ' + e.toString());
      return null;
    }
  }
  
  /**
   * Store model coefficients in script properties
   * 
   * @param {Object} coefficients - Model coefficients to store
   * @returns {boolean} Success status
   */
  function storeModelCoefficients(coefficients) {
    try {
      const props = PropertiesService.getScriptProperties();
      props.setProperty('PREDICTION_MODEL_COEFFICIENTS', JSON.stringify(coefficients));
      return true;
    } catch (e) {
      Logger.log('Error storing model coefficients: ' + e.toString());
      return false;
    }
  }
  
  /**
   * Train a new logistic regression model using historical data
   * 
   * @returns {Object} Trained model coefficients
   */
  function trainModel() {
    // Define a default model based on domain knowledge for when we don't have enough data
    // These weights are initial estimates and will be refined as more data becomes available
    const defaultModelCoefficients = {
      version: '1.0',
      trainedAt: new Date().toISOString(),
      weights: {
        intercept: 0.2,            // Baseline
        qualityScore: 0.5,         // Higher quality score increases win probability
        previousResult: -0.15,     // After win is slightly negative, after loss is positive
        liquidityResistance: -0.3, // Higher resistance decreases win probability
        priceClarity: -0.3,        // Less clarity decreases win probability
        tradeCount: -0.1,          // Higher trade count slightly decreases probability
        marketVolatility: 0.1,     // Some volatility can be good
        riskReward: 0.45,          // Better risk/reward ratio increases win probability
        timeOfDay: 0.1,            // Time of day effect
        dayOfWeek: 0.05            // Day of week effect
      },
      metadata: {
        dataPoints: 0,
        accuracy: 0.6,  // Estimated accuracy
        lastUpdated: new Date().toISOString()
      }
    };
    
    try {
      // Attempt to get historical trades from Pattern Analysis History
      const ss = SpreadsheetApp.getActiveSpreadsheet();
      const analysisSheet = ss.getSheetByName('Pattern Analysis History');
      
      if (!analysisSheet) {
        // No data available, return default model
        Logger.log('No Pattern Analysis History sheet found. Using default model.');
        storeModelCoefficients(defaultModelCoefficients);
        return defaultModelCoefficients;
      }
      
      // Get all data
      const data = analysisSheet.getDataRange().getValues();
      
      // Skip header
      const rows = data.slice(1);
      
      // Filter to only include rows with results
      const completedPredictions = rows.filter(row => row[12]);
      
      if (completedPredictions.length < 10) {
        // Not enough data yet, return default model with a note
        Logger.log(`Only ${completedPredictions.length} completed predictions found. Need at least 10 for training. Using default model.`);
        defaultModelCoefficients.metadata.dataPoints = completedPredictions.length;
        storeModelCoefficients(defaultModelCoefficients);
        return defaultModelCoefficients;
      }
      
      // With more data, we could implement actual logistic regression training here
      // For now, we'll adjust the default model weights based on the patterns we see in the data
      
      // This is a simplified approach to learning from the data
      // In a real implementation, we would use proper gradient descent to optimize weights
      const learnedWeights = {...defaultModelCoefficients.weights};
      let correctPredictions = 0;
      
      completedPredictions.forEach(row => {
        // Extract features
        const features = {
          qualityScore: row[3] || 3.5,
          previousResult: row[4] || 'none',
          liquidityResistance: row[5] || 'low',
          priceClarity: row[6] || 'clear',
          tradeCount: row[7] || 1,
          marketVolatility: row[8] || 'normal',
          riskReward: row[9] || 'good'
        };
        
        // Normalize features
        const normalizedFeatures = normalizeFeatures(features);
        
        // Get actual result
        const actualResult = row[12].toLowerCase() === 'win' ? 1 : 0;
        
        // Calculate predicted probability using current weights
        const predictedProb = calculateProbability(normalizedFeatures, {weights: learnedWeights});
        const predictedResult = predictedProb >= 0.5 ? 1 : 0;
        
        // Check if prediction was correct
        if (predictedResult === actualResult) {
          correctPredictions++;
        }
        
        // Very simple weight adjustment based on prediction error
        // In a real implementation, we would use proper gradient descent
        const error = actualResult - predictedProb;
        const learningRate = 0.01; // Small learning rate to make gradual adjustments
        
        // Update weights based on prediction error
        Object.keys(normalizedFeatures).forEach(feature => {
          learnedWeights[feature] += learningRate * error * normalizedFeatures[feature];
        });
      });
      
      // Calculate accuracy
      const accuracy = correctPredictions / completedPredictions.length;
      
      // Create and store the trained model
      const trainedModel = {
        version: '1.1',
        trainedAt: new Date().toISOString(),
        weights: learnedWeights,
        metadata: {
          dataPoints: completedPredictions.length,
          accuracy: accuracy,
          lastUpdated: new Date().toISOString()
        }
      };
      
      storeModelCoefficients(trainedModel);
      Logger.log(`Model trained on ${completedPredictions.length} data points with accuracy ${accuracy.toFixed(2)}`);
      
      return trainedModel;
    } catch (e) {
      Logger.log('Error training model: ' + e.toString());
      
      // Return default model on error
      storeModelCoefficients(defaultModelCoefficients);
      return defaultModelCoefficients;
    }
  }
  
  /**
   * Evaluate model performance on historical data
   * 
   * @returns {Object} Performance metrics
   */
  function evaluateModel() {
    try {
      const ss = SpreadsheetApp.getActiveSpreadsheet();
      const analysisSheet = ss.getSheetByName('Pattern Analysis History');
      
      if (!analysisSheet) {
        return {
          accuracy: 0,
          dataPoints: 0,
          error: 'No Pattern Analysis History sheet found'
        };
      }
      
      // Get model coefficients
      const coefficients = getModelCoefficients() || trainModel();
      
      if (!coefficients || !coefficients.weights) {
        return {
          accuracy: 0,
          dataPoints: 0,
          error: 'Unable to load or train prediction model'
        };
      }
      
      // Get all data
      const data = analysisSheet.getDataRange().getValues();
      
      // Skip header
      const rows = data.slice(1);
      
      // Filter to only include rows with results
      const completedPredictions = rows.filter(row => row[12]);
      
      if (completedPredictions.length === 0) {
        return {
          accuracy: 0,
          dataPoints: 0,
          error: 'No completed predictions found'
        };
      }
      
      // Evaluate each prediction
      let correctPredictions = 0;
      let totalError = 0;
      
      completedPredictions.forEach(row => {
        // Extract features
        const features = {
          qualityScore: row[3] || 3.5,
          previousResult: row[4] || 'none',
          liquidityResistance: row[5] || 'low',
          priceClarity: row[6] || 'clear',
          tradeCount: row[7] || 1,
          marketVolatility: row[8] || 'normal',
          riskReward: row[9] || 'good'
        };
        
        // Normalize features
        const normalizedFeatures = normalizeFeatures(features);
        
        // Get actual result
        const actualResult = row[12].toLowerCase() === 'win' ? 1 : 0;
        
        // Calculate predicted probability using current weights
        const predictedProb = calculateProbability(normalizedFeatures, coefficients);
        const predictedResult = predictedProb >= 0.5 ? 1 : 0;
        
        // Check if prediction was correct
        if (predictedResult === actualResult) {
          correctPredictions++;
        }
        
        // Calculate squared error for this prediction
        totalError += Math.pow(actualResult - predictedProb, 2);
      });
      
      // Calculate metrics
      const accuracy = correctPredictions / completedPredictions.length;
      const mse = totalError / completedPredictions.length; // Mean Squared Error
      
      return {
        accuracy: accuracy,
        mse: mse,
        dataPoints: completedPredictions.length,
        modelVersion: coefficients.version || '1.0'
      };
    } catch (e) {
      Logger.log('Error evaluating model: ' + e.toString());
      return {
        accuracy: 0,
        dataPoints: 0,
        error: e.toString()
      };
    }
  }
  
  /**
   * Creates a UI for analyzing a potential trade

  /**
   * Shows a dialog with model performance metrics
   */
  function showModelPerformance() {
    const metrics = evaluateModel();
    
    // Format the metrics for display
    let html = '<div style="padding: 15px; font-family: Arial, sans-serif;">';
    html += '<h2>Trade Prediction Model Performance</h2>';
    
    if (metrics.error) {
      html += `<p style="color: red;">Error: ${metrics.error}</p>`;
    } else {
      html += `<p><strong>Model Version:</strong> ${metrics.modelVersion || 'Unknown'}</p>`;
      html += `<p><strong>Accuracy:</strong> ${(metrics.accuracy * 100).toFixed(1)}%</p>`;
      html += `<p><strong>Data Points:</strong> ${metrics.dataPoints}</p>`;
      
      if (metrics.mse !== undefined) {
        html += `<p><strong>Mean Squared Error:</strong> ${metrics.mse.toFixed(4)}</p>`;
      }
      
      // Get coefficients for inspection
      const coefficients = getModelCoefficients();
      if (coefficients && coefficients.weights) {
        html += '<h3>Model Weights</h3>';
        html += '<table style="width: 100%; border-collapse: collapse;">';
        html += '<tr><th style="text-align: left; padding: 5px; border-bottom: 1px solid #ddd;">Feature</th><th style="text-align: right; padding: 5px; border-bottom: 1px solid #ddd;">Weight</th></tr>';
        
        Object.keys(coefficients.weights).forEach(feature => {
          const weight = coefficients.weights[feature];
          const style = weight > 0 ? 'color: green;' : (weight < 0 ? 'color: red;' : '');
          html += `<tr><td style="padding: 5px; border-bottom: 1px solid #ddd;">${feature}</td><td style="text-align: right; padding: 5px; border-bottom: 1px solid #ddd; ${style}">${weight.toFixed(3)}</td></tr>`;
        });
        
        html += '</table>';
      }
    }
    
    html += '</div>';
    
    const htmlOutput = HtmlService.createHtmlOutput(html)
      .setWidth(400)
      .setHeight(500)
      .setTitle('Model Performance');
    
    SpreadsheetApp.getUi().showModalDialog(htmlOutput, 'Model Performance');
